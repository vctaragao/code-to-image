package main

import "fmt"

func main() {
    resp, err := makeRequest()
    if err != nil {
        fmt.Printf("Error: %v\n", err)
        return
    }

    dto, err := parseResponse(resp)
    if err != nil {
        fmt.Printf("Error: %v\n", err)
        return
    }

    err = saveQuotation(dto)
    if err != nil {
        fmt.Printf("Error: %v\n", err)
        return
    }
}
